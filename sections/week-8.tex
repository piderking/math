}

\problem[Group 6. Sec 4.1 3]{
	Balloon rises vertically at rate of 10 ft/sec. A person watches the balloon ascend from a point on the groud
	100ft away from the spot below the rising balloon. At what rate (radians/second) is the observer's eye rotating upward to follow the ballon when it is 50ft above the level of the observer's eye

}
\step[Definitions]{
	$$\frac{dh}{dt}=10$$
	$$\frac{db}{dt}=0$$
	$$ b = 100 $$
	$$ h = 50 $$

	$$tan(\theta) = \frac{h}{b}$$
}

\step[Setting Up]{
	$$\theta = tan^{-1}(\frac{h}{b})$$
	$$u = \frac{h}{b}$$
	$$\frac{du}{dt} = \frac{\frac{dh}{dt}b - h\frac{db}{dt}}{b^2}$$


	$$\frac{du}{dt} = \frac{\frac{dh}{dt}b }{b^2}$$
	
	$$\frac{du}{dt} = \frac{\frac{dh}{dt} }{b}$$
	$$\frac{du}{dt} = \frac{1}{10}$$

	$$u = \frac{50}{100}$$
	$$u = \frac{1}{2}$$


	$$\frac{d\theta}{dt} = \frac{\frac{du}{dt}}{ 1 + u^2}$$
}

\step[Plugging In]{
	
	$$\frac{d\theta}{dt} = \frac{\frac{1}{10}}{ \frac{5}{4} }$$

	$$\frac{d\theta}{dt} = \frac{1}{10} * \frac{4}{5} $$


}
\solution{

	$$\frac{d\theta}{dt} = \frac{4}{50} $$
}

\problem[\# 3]{
A ladder 12 feet long is leaning against a vertical wall with its other end on the
ground. The top end of the ladder is sliding down the wall. When the top end is 7
feet from the ground, it is sliding down the wall at 2 ft/sec. How fast is the bottom
moving away from the wall at this instant?
}

\step[Diagram]{
% Example 1: Full labels, using key-value syntax
\righttriangle{4}{3}{(0,0)}[
    label A={}, label B={}, label C={},
	    side a={$l=12$}, side b={h=7}, side c={b},
    % angle alpha=90^\circ, angle beta=\theta
]
}
 \step[Definitions]{
	 Sliding Away
	 $$\frac{dh}{dt} = -2$$
	 Ladder Length stays the same
	 $$\frac{dl}{dt} = 0$$

	 Relationship of Sides
	 $$b^2 + h^2 = l^2$$
	 $$b^2 = l^2 - h^2$$

	 Length of b
	 $$b = \sqrt{144 - 49}$$
	 $$b = \sqrt{95}$$
 }

 \step[Derivative]{
	 $$2b * \frac{db}{dt} = 2l * \frac{dl}{dt} - 2h * \frac{dh}{dt}$$
 
	 $$\frac{db}{dt} = \frac{ 2l * \frac{dl}{dt} - 2h * \frac{dh}{dt}}{2\sqrt{95}}$$


	 $$\frac{db}{dt} = \frac{  0 - (14 * -2) }{2\sqrt{95}}$$


	 $$\frac{db}{dt} = \frac{  14 }{\sqrt{95}}$$
 }

 \solution{
The bottom is sliding at $\frac{  14 }{\sqrt{95}}$ or  $1.43636969$ ft/sec
 }
